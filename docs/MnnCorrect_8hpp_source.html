262 // Function name decays to a function pointer, should be callable by just doing builder().
263 auto builder = (approximate ? approximate_builder : exact_builder);
264
265 if (order == NULL) {
266 if (automatic_order) {
267 AutomaticOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors, reference_policy, nthreads);
268 runner.run(num_mads, robust_iterations, robust_trim);
269 return Details(runner.get_order(), runner.get_num_pairs());

365 template<typename Batch>
366 Details run(int ndim, size_t nobs, const Float* input, const Batch* batch, Float* output, const int* order = NULL) {
367 const Batch nbatches = (nobs ? *std::max_element(batch, batch + nobs) + 1 : 0);
368 std::vector<size_t> sizes(nbatches);
369 for (size_t o = 0; o < nobs; ++o) {
370 ++sizes[batch[o]];
371 }
372
373 // Avoiding the need to allocate a temporary buffer
374 // if we're already dealing with contiguous batches.
375 bool already_sorted = true;
376 for (size_t o = 1; o < nobs; ++o) {
377 if (batch[o] < batch[o-1]) {
378 already_sorted = false;
379 break;
380 }
381 }
382 if (already_sorted) {
383 return run(ndim, sizes, input, output, order);
384 }
385
386 size_t accumulated = 0;
387 std::vector<size_t> offsets(nbatches);
388 for (size_t b = 0; b < nbatches; ++b) {
389 offsets[b] = accumulated;
390 accumulated += sizes[b];
391 }
392
393 // Dumping everything by order into another vector.
394 std::vector<Float> tmp(ndim * nobs);
395 std::vector<const Float*> ptrs(nbatches, tmp.data());
396 for (size_t b = 0; b < nbatches; ++b) {
397 ptrs[b] += offsets[b] * ndim;
398 }
399 for (size_t o = 0; o < nobs; ++o) {
400 auto current = input + o * ndim;
401 auto& offset = offsets[batch[o]];
402 auto destination = tmp.data() + ndim * offset;
403 std::copy(current, current + ndim, destination);
404 ++offset;
405 }
406
407 auto stats = run_internal(ndim, sizes, ptrs, output, order);
408 restore_order(ndim, stats.merge_order, sizes, batch, output);
409 return stats;
410 }
