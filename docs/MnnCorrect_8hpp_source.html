212 Details run_internal(int ndim, const std::vector<size_t>& nobs, const std::vector<const Float*>& batches, Float* output, const int* order) {
213 // Function name decays to a function pointer, should be callable by just doing builder().
214 auto builder = (approximate ? approximate_builder : exact_builder);
215
216 if (order == NULL) {
217 if (automatic_order) {
218 AutomaticOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors);
219 runner.run(num_mads, robust_iterations, robust_trim);
220 return Details(runner.get_order(), runner.get_num_pairs());
221 } else {
222 std::vector<int> trivial_order(nobs.size());
223 std::iota(trivial_order.begin(), trivial_order.end(), 0);
224 CustomOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors, trivial_order.data());
225 runner.run(num_mads, robust_iterations, robust_trim);
226 return Details(std::move(trivial_order), runner.get_num_pairs());
227 }
228 } else {
229 CustomOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors, order);
230 runner.run(num_mads, robust_iterations, robust_trim);

316 template<typename Batch>
317 Details run(int ndim, size_t nobs, const Float* input, const Batch* batch, Float* output, const int* order = NULL) {
318 const Batch nbatches = (nobs ? *std::max_element(batch, batch + nobs) + 1 : 0);
319 std::vector<size_t> sizes(nbatches);
320 for (size_t o = 0; o < nobs; ++o) {
321 ++sizes[batch[o]];
322 }
323
324 // Avoiding the need to allocate a temporary buffer
325 // if we're already dealing with contiguous batches.
326 bool already_sorted = true;
327 for (size_t o = 1; o < nobs; ++o) {
328 if (batch[o] < batch[o-1]) {
329 already_sorted = false;
330 break;
331 }
332 }
333 if (already_sorted) {
334 return run(ndim, sizes, input, output, order);
335 }
336
337 size_t accumulated = 0;
338 std::vector<size_t> offsets(nbatches);
339 for (size_t b = 0; b < nbatches; ++b) {
340 offsets[b] = accumulated;
341 accumulated += sizes[b];
342 }
343
344 // Dumping everything by order into another vector.
345 std::vector<Float> tmp(ndim * nobs);
346 std::vector<const Float*> ptrs(nbatches, tmp.data());
347 for (size_t b = 0; b < nbatches; ++b) {
348 ptrs[b] += offsets[b] * ndim;
349 }
350 for (size_t o = 0; o < nobs; ++o) {
351 auto current = input + o * ndim;
352 auto& offset = offsets[batch[o]];
353 auto destination = tmp.data() + ndim * offset;
354 std::copy(current, current + ndim, destination);
355 ++offset;
356 }
357
358 auto stats = run_internal(ndim, sizes, ptrs, output, order);
359 restore_order(ndim, stats.merge_order, sizes, batch, output);
360 return stats;
361 }
