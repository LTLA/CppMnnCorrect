232 Details run_internal(int ndim, const std::vector<size_t>& nobs, const std::vector<const Float*>& batches, Float* output, const int* order) {
233 // Function name decays to a function pointer, should be callable by just doing builder().
234 auto builder = (approximate ? approximate_builder : exact_builder);
235
236 if (order == NULL) {
237 if (automatic_order) {
238 AutomaticOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors, reference_policy);
239 runner.run(num_mads, robust_iterations, robust_trim);
240 return Details(runner.get_order(), runner.get_num_pairs());
241 } else {
242 std::vector<int> trivial_order(nobs.size());
243 std::iota(trivial_order.begin(), trivial_order.end(), 0);
244 CustomOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors, trivial_order.data());
245 runner.run(num_mads, robust_iterations, robust_trim);
246 return Details(std::move(trivial_order), runner.get_num_pairs());
247 }
248 } else {
249 CustomOrder<Index, Float, decltype(builder)> runner(ndim, nobs, batches, output, builder, num_neighbors, order);
250 runner.run(num_mads, robust_iterations, robust_trim);

336 template<typename Batch>
337 Details run(int ndim, size_t nobs, const Float* input, const Batch* batch, Float* output, const int* order = NULL) {
338 const Batch nbatches = (nobs ? *std::max_element(batch, batch + nobs) + 1 : 0);
339 std::vector<size_t> sizes(nbatches);
340 for (size_t o = 0; o < nobs; ++o) {
341 ++sizes[batch[o]];
342 }
343
344 // Avoiding the need to allocate a temporary buffer
345 // if we're already dealing with contiguous batches.
346 bool already_sorted = true;
347 for (size_t o = 1; o < nobs; ++o) {
348 if (batch[o] < batch[o-1]) {
349 already_sorted = false;
350 break;
351 }
352 }
353 if (already_sorted) {
354 return run(ndim, sizes, input, output, order);
355 }
356
357 size_t accumulated = 0;
358 std::vector<size_t> offsets(nbatches);
359 for (size_t b = 0; b < nbatches; ++b) {
360 offsets[b] = accumulated;
361 accumulated += sizes[b];
362 }
363
364 // Dumping everything by order into another vector.
365 std::vector<Float> tmp(ndim * nobs);
366 std::vector<const Float*> ptrs(nbatches, tmp.data());
367 for (size_t b = 0; b < nbatches; ++b) {
368 ptrs[b] += offsets[b] * ndim;
369 }
370 for (size_t o = 0; o < nobs; ++o) {
371 auto current = input + o * ndim;
372 auto& offset = offsets[batch[o]];
373 auto destination = tmp.data() + ndim * offset;
374 std::copy(current, current + ndim, destination);
375 ++offset;
376 }
377
378 auto stats = run_internal(ndim, sizes, ptrs, output, order);
379 restore_order(ndim, stats.merge_order, sizes, batch, output);
380 return stats;
381 }
