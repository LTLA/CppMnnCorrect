286 template<typename Batch>
287 Details run(int ndim, size_t nobs, const Float* input, const Batch* batch, Float* output) {
288 const Batch nbatches = (nobs ? *std::max_element(batch, batch + nobs) + 1 : 0);
289 std::vector<size_t> sizes(nbatches);
290 for (size_t o = 0; o < nobs; ++o) {
291 ++sizes[batch[o]];
292 }
293
294 // Avoiding the need to allocate a temporary buffer
295 // if we're already dealing with contiguous batches.
296 bool already_sorted = true;
297 for (size_t o = 1; o < nobs; ++o) {
298 if (batch[o] < batch[o-1]) {
299 already_sorted = false;
300 break;
301 }
302 }
303 if (already_sorted) {
304 return run(ndim, sizes, input, output);
305 }
306
307 size_t accumulated = 0;
308 std::vector<size_t> offsets(nbatches);
309 for (size_t b = 0; b < nbatches; ++b) {
310 offsets[b] = accumulated;
311 accumulated += sizes[b];
312 }
313
314 // Dumping everything by order into another vector.
315 std::vector<Float> tmp(ndim * nobs);
316 std::vector<const Float*> ptrs(nbatches, tmp.data());
317 for (size_t b = 0; b < nbatches; ++b) {
318 ptrs[b] += offsets[b] * ndim;
319 }
320 for (size_t o = 0; o < nobs; ++o) {
321 auto current = input + o * ndim;
322 auto& offset = offsets[batch[o]];
323 auto destination = tmp.data() + ndim * offset;
324 std::copy(current, current + ndim, destination);
325 ++offset;
326 }
327
328 auto stats = run_internal(ndim, sizes, ptrs, output);
329 restore_order(ndim, stats.merge_order, sizes, batch, output);
330 return stats;
331 }
