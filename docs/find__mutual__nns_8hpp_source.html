46template<typename Index, typename Float>
47MnnPairs<Index> find_mutual_nns(const NeighborSet<Index, Float>& left, const NeighborSet<Index, Float>& right) {
48 Index nleft = left.size();
49 Index nright = right.size();
50
51 std::vector<std::vector<Index> > neighbors_of_left(nleft);
52 for (Index l = 0; l < nleft; ++l) {
53 auto& storage = neighbors_of_left[l];
54 for (const auto& f : left[l]) {
55 storage.push_back(f.first);
56 }
57 std::sort(storage.begin(), storage.end());
58 }
59
60 MnnPairs<Index> output(nright);
61 std::vector<size_t> last(nleft);
62 for (Index r = 0; r < nright; ++r) {
63 const auto& mine = right[r];
64 std::vector<Index> holder;
65
66 for (auto left_pair : mine) {
67 auto left_neighbor = left_pair.first;
68 const auto& other = neighbors_of_left[left_neighbor];
69 auto& position = last[left_neighbor];
70
71 for (; position < other.size(); ++position) {
72 if (other[position] >= r) {
73 if (other[position] == r) {
74 holder.push_back(left_neighbor);
75 ++output.num_pairs;
76 }
77 break;
78 }
79 }
80 }
81
82 if (holder.size()) {
83 output.matches[r] = std::move(holder);
84 }
85 }
86
87 return output;
88}
